import { Tabs, TabItem, Code } from "@astrojs/starlight/components";
import Button from "@bambiui/components/button/astro";

## Installation

```sh
npx bambiui init
npx bambiui add button
```

The CLI copies the Button source into your project. The installed files are self-contained and do not depend on a runtime bambiui component package.

Import the token file once from your global stylesheet or app entry:

```css
@import "./styles/bambi.css";
```

Then import the component:

<Tabs syncKey="framework">
  <TabItem label="React">
    ```tsx
    import { Button } from './components/ui/button';
    ```
  </TabItem>
  <TabItem label="Svelte">

  </TabItem>
  <TabItem label="Astro">

  </TabItem>
</Tabs>

## API Reference

| Prop         | Type                                                             | Default     | Description                            |
| ------------ | ---------------------------------------------------------------- | ----------- | -------------------------------------- |
| `intent`     | `'primary' \| 'secondary' \| 'danger' \| 'success' \| 'warning'` | `'primary'` | Semantic meaning                       |
| `appearance` | `'solid' \| 'outline' \| 'ghost' \| 'link'`                      | `'solid'`   | Visual treatment                       |
| `size`       | `'sm' \| 'md' \| 'lg' \| 'icon'`                                 | `'md'`      | Button size                            |
| `loading`    | `boolean`                                                        | `false`     | Shows spinner and prevents interaction |
| `disabled`   | `boolean`                                                        | `false`     | Disables the button                    |
| `type`       | `'button' \| 'submit' \| 'reset'`                                | `'button'`  | HTML button type                       |

All other HTML button attributes (`aria-*`, `style`, `class`) are forwarded to the underlying `<button>` element. React uses `className`; Svelte, Vue, and Astro use `class`.
